    def _cmpcheck(self, op, other, default=False):
        if type(self) == type(other):
            return op(self._cmpkey(), other._cmpkey())
        else:
            return default


[docs]    def __eq__(self, other):
        if self is other:
            return True
        else:
            return self._cmpcheck(ops.__eq__, other, default=False)
